面试题Python编程题

面试真题：

1. 编写函数,实现功能：将[1,2,[3,[4,5]],6,[7,]] 转换成[1,2,3,4,5,6,7]

li = [1, 2, [3, [4, 5, [1, 2, [1, 2]]]], 6, [7, ]]

方法一

def func( li ):
    # lis = []
    if isinstance(li, list):
        return [y for x in li for y in func(x)]
    else:
        return [li]

print(func(li))

方法二

lis = []
def func(li):
    for i in li:
        if isinstance(i,list):
           func(i)
        else:
            lis.append(i)
    return lis

print(list(func(li)))

2. [1,2,[3,[4,5]],6,[7,]] 用生成器将其生成[1,2,3,4,5,6,7]

生成器 一

def func3(li):
    for i in li:
        if isinstance(i, list):
            for j in func3(i):
                yield j
        else:
            yield i
print(list(func3(li)))

生产器yield from

def func4(li):
    for i in li:
        if isinstance(i, list):
            yield from func4(i)
        else:
            yield i
print(list(func4(li)))
